### 2.2 `init-project` final contract

```
mdagent init-project [--force] [--no-install-skills] [--json] DIR
```

  - Refuse if `DIR` exists, is non-empty, and `--force` is not set.
  - Materialize every file from the bundled kit at the corresponding
    relative path. Files with a `mode` in the manifest get
    `os.chmod(dest, mode)`.
  - Always write `DIR/MANIFEST.json` last (= a marker the materialization
    completed cleanly).
  - Unless `--no-install-skills`, also call the shared
    `_install_skills_core(dest_root=DIR, dry_run=False)` to populate
    `DIR/.claude/skills/`.
  - `--json` returns one combined payload:
    `{action: "init-project", target: "...", manifest: {...}, install_skills: {...}}`.

### 2.3 verify.sh final form

```bash
#!/usr/bin/env bash
set -e

REPO_ROOT="$(cd "$(dirname "$0")" && pwd)"

# 1. mdagent installed?
command -v mdagent >/dev/null 2>&1 || { echo "mdagent not found on PATH."; exit 1; }
mdagent --version

# 2. Doctor (gmx + network optional)
mdagent doctor --gmx-required --check-network

# 3. Structural — every file the kit needs is here.
for f in \
  README.md \
  .gitignore \
  structures/1aki.pdb \
  run_configs/lysozyme_short.json \
  run_configs/lysozyme_rcsb_tutorial.json \
  run_configs/general_md_prep_example.json \
  tutorial/getting_started.md \
  runs/.gitkeep \
  .claude/skills/md-run-workflow/SKILL.md \
  .claude/skills/md-prep-structure/SKILL.md \
  .claude/skills/md-visualize/SKILL.md \
  MANIFEST.json
do
  [[ -f "$REPO_ROOT/$f" ]] || { echo "missing: $f"; exit 1; }
done

# 4. Config validity — schema-validate each config via mdagent.
for c in "$REPO_ROOT"/run_configs/*.json; do
  python3 -c "
import sys, json
from mdagent import RunConfig
RunConfig.from_file(sys.argv[1])
print('  config OK:', sys.argv[1])
" "$c" || { echo "config invalid: $c"; exit 1; }
done

if [[ "${1:-}" == "--run-smoke" ]]; then
  cd "$REPO_ROOT"
  rm -rf runs/smoke
  mdagent run-workflow --runs-root ./runs --config ./run_configs/lysozyme_short.json --run-id smoke
  grep -q "readiness: \*\*ready\*\*" runs/smoke/REPORT.md || { echo "smoke run did not report ready"; exit 1; }
  echo "✓ smoke run produced a 'ready' REPORT"
fi
echo "✓ starter kit verified"
```

```python
def test_starter_kit_init_and_smoke(tmp_path):
    # 1) Build wheel + install in clean venv
    ...
    # 2) Run mdagent init-project from a directory with NO checkout
    project = tmp_path / "newproj"
    subprocess.run([binexe, "init-project", str(project)], check=True)
    # 3) Assert manifest covers every file
    manifest = json.load((project / "MANIFEST.json").open())
    for entry in manifest["files"]:
        assert (project / entry["path"]).is_file()
    # 4) Assert .claude/skills/ has all three
    skills = sorted(p.name for p in (project / ".claude" / "skills").iterdir())
    assert skills == ["md-prep-structure", "md-run-workflow", "md-visualize"]
    # 5) verify.sh structural mode
    r = subprocess.run([str(project / "verify.sh")], capture_output=True, text=True)
    assert r.returncode == 0, r.stderr
    # 6) verify.sh --run-smoke (if gmx is present)
    if shutil.which("gmx"):
        r = subprocess.run([str(project / "verify.sh"), "--run-smoke"],
                           capture_output=True, text=True, timeout=600)
        assert r.returncode == 0, r.stderr
        assert "readiness: **ready**" in (project / "runs" / "smoke" / "REPORT.md").read_text()
    # 7) Re-init refuses
    r = subprocess.run([binexe, "init-project", str(project)], capture_output=True, text=True)
    assert r.returncode != 0
    # 8) --force succeeds
    subprocess.run([binexe, "init-project", "--force", str(project)], check=True)
```
